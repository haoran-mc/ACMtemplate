long long binPow(long long base, long long expo) {
    long long res = 1;
    while (expo != 0) {
        if (expo & 1)
            res = (1LL * res * base) % mod;

        base = (1LL * base * base) % mod;
        expo >>= 1;
    }
    return res;
}

long long inv(long long x) {
    return binPow(x, mod-2, mod);
}

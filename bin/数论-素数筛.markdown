### 欧拉筛法

```cpp
#include <cstdio>
const int maxn = 5e8;
int seek = 0;
int prime[maxn];
bool sifter[maxn];
//bool* sifter = (bool *)memset(sifter, 0, sizeof(sifter));

void sievePrime() {
    for (int i = 2; i <= maxn; i++) {
        if (sifter[i] == false)
            prime[seek++] = i;

        for (int j = 0; j < seek; j++) {
            if (i * prime[j] > maxn) break;

            sifter[i * prime[j]] = true;

            if (!(i % prime[j]))  break;
        }
    }
}

int main()
{
    sievePrime();
    for (int i = 0; i < seek; i++)
        printf("%d ", prime[i]);
    return 0;
}
```

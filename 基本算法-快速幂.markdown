#include <cstdio>
const int mod = 99991;

template<typename T>
T binPow(T base, T expo) {
    T res = 1;
    while (expo != 0) {
        if (expo & 1)
            res = (1LL * res * base) % mod;

        base = (1LL * base * base) % mod;
        expo >>= 1;
    }
    return res;
}

int main()
{
    long long base = 23, expo = 19898283988388888;
    long long res;
    for (int i = 0; i < 1000000; ++i)
        res = binPow(base, expo);

    printf("%lld\n", res);
    return 0;
}
